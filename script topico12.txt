# 2. Crie uma nova base de dados contendo duas tabelas com pelo menos três colunas cada.
create database dinossauro;

use dinossauro;


CREATE TABLE especies (
    id_especie INT PRIMARY KEY AUTO_INCREMENT,
    nome_cientifico VARCHAR(100) NOT NULL,
    periodo_geologico VARCHAR(50),
    dieta ENUM('Carnívoro', 'Herbívoro', 'Onívoro')
);

CREATE TABLE fosseis (
    id_fossil INT PRIMARY KEY AUTO_INCREMENT,
    id_especie INT,
    local_descoberta VARCHAR(100)
);

# 3. Crie um novo usuário chamado **programador**.
create user 'programador'@'localhost' identified by 'python';

# 4. Conceda ao usuário **programador** apenas os privilégios de manipulação de dados (**SELECT**, **INSERT** e **UPDATE**) em apenas uma das tabelas criadas.
grant SELECT, INSERT, UPDATE ON dinossauro.* TO 'programador'@'localhost';
